# N皇后问题

## 问题分析

一个著名的特例就是八皇后问题，这是回溯算法的典型案例。

**回溯算法**



## 代码

class Solution {
    private static int DOT = 46;
    private static int QUEEN = 81;
    public static void main(String[] args) {
        Solution solution = new Solution();
        solution.solveNQueens(4).forEach(board -> {
            board.forEach(System.out::println);
            System.out.println("========");
        });
    }

    public List<List<String>> solveNQueens(int n) {
        List<List<String>> ans = new ArrayList<>();
        int[] queens = new int[n];
        backTrack(ans, queens, 0);
        return ans;
    }

    public void backTrack(List<List<String>> ans, int[] queens, int row) {
        if (row == queens.length) {
            List<String> board = new ArrayList<>(row);
            for (int queen: queens) {
                board.add(printRow(queen, row));
            }
            ans.add(board);
            return;
        }
        for (int col = 0; col < queens.length; col++) {
            if (attemptToPut(queens, row, col)){
                backTrack(ans, queens, row + 1);
            }
        }
    }

    public boolean attemptToPut(int[] queens, int row, int col) {
        for (int i = 0; i < row; i++) {
            if (queens[i] == col) {
                return false;
            }
            if (Math.abs(row - i) == Math.abs(col - queens[i])) {
                return false;
            }
        }
        queens[row] = col;
        return true;
    }

    public String printRow(int col, int n) {
        char[] chars = new char[n];
        Arrays.fill(chars, '.');
        chars[col] = 'Q';
        return new String(chars);
    }

}
